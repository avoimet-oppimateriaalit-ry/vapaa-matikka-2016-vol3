\begin{tehtava}
	Olkoon $f(x)=x^5$ ja $g(x)=x^3-2x$. Määritä raja-arvot.
	\alakohdat{
		§ $\lim\limits_{x\to2} f(x)$
		§ $\lim\limits_{x\to2} g(x)$
		§ $\lim\limits_{x\to2} (f+g)(x)$
		§ $\lim\limits_{x\to2} (fg)(x)$
	}
	\begin{vastaus}
		\alakohdat{
			§ $32$
			§ $4$
			§ $36$
			§ $128$
		}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Funktion $f(x)$ raja-arvo pisteessä $x=1$ on $5$. Määritellään lisäksi $g(x) = x^2+3$. Mikä on funktion $h(x) = 4f(x) + 3g(x)$ raja-arvo pisteessä $x=1$?
	\begin{vastaus}
		$32$
	\end{vastaus}
\end{tehtava}
