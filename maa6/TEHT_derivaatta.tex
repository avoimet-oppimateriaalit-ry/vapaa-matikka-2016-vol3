\begin{tehtava}
	Laske funktioiden derivaatat pisteessä $a = 0$.
	\alakohdat{
		§ $f_1(x)=x$
		§ $f_2(x)=x^2$
		§ $f_3(x)=x \cdot 2^x$
		§ $f_4(x)=\frac1{x^2+1}$
	}
	\begin{vastaus}
		\alakohdat{
			§ $f'_1(0)=1$
			§ $f'_2(0)=0$
			§ $f'_3(0)=1$
			§ $f'_4(0)=0$
		}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Onko itseisarvofunktiolla $|x|$ derivaattaa pisteessä $x=0$?
	\begin{vastaus}
		Ei.
	\end{vastaus}
\end{tehtava}

\begin{tehtava}

		\luettelo{

		}

\end{tehtava}

\begin{tehtava}

\begin{tehtava}

\begin{tehtava} % pohjustaa MAA7:ää
	Oletetaan, että tunnemme seuraavat:
	\luettelo{
		§ $f(x) = \sin x + 2 \cos x$, $f'(x) = -2 \sin x + \cos x$
		§ $g(x) = 5 \sin x - 2 \cos x$, $g'(x) = 2 \sin x + 5 \cos x$
	}
	Määritä näiden perusteella funktioiden $\sin x$ ja $\cos x$ derivaattafunktiot käyttäen tunnettuja derivaattojen laskusääntöjä. Trigonometristen funktioiden derivaattoihin syvennytään kurssissa MAA7. % OPS 2016
	\begin{vastaus}
		\luettelo{
			§ $\diff{\sin x} = \cos x$
			§ $\diff{\cos x} = -\sin x$
		}
	\end{vastaus}
\end{tehtava}

\begin{tehtava} % haastavahko rutiininparannustehtävä
	Määritellään ns. $xD$-operaattori: funktiosta otetaan ensin derivaatta ja sen jälkeen saatu lauseke kerrotaan $x$:llä. Esimerkiksi $(xD)(x^3) = 3x^3$ ja $(xD)^2 (x^3) = 9x^3$. Laske.
	\alakohdat{
		§ $(xD) (x^5+4x+1)$
		§ $(xD) (x^6+3x^3+22x^2+10)$
		§ $(xD)^2 (x^6+3x^3+22x^2+10)$
		§ $(xD) (3x^6+7x^5+4x^2+43x)$
		§ $(xD)^2 (3x^6+7x^5+4x^2+43x)$
		§ $(xD)^2 (7x^2+32x)$
		§ $(xD)^3 (7x^2+32x)$
	}
	\begin{vastaus}
		\alakohdat{
			§ $5x^5+4x$
			§ $6x^6+9x^3+44x^2$
			§ $36x^6+27x^3+88x^2$
			§ $18x^6+35x^5+8x^2+43x$
			§ $108x^6+175x^5+16x^2+43x$
			§ $28x^2+32x$
			§ $56x^2+32x$
		}		
	\end{vastaus}
\end{tehtava}
